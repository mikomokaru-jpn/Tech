//----------------------------------------------------------------------
//文字列のトリミング：全角空白も削除される
//----------------------------------------------------------------------
inText = "　 abc   def　 "; 
console.log(inText); 

//----------------------------------------------------------------------
//文字列を空白文字で分割する：複数の空白文字は正規表現で指定する
//----------------------------------------------------------------------
inText = inText.trim();
const re = /\s+/;
let keywords = inText.split(re);
console.log(keywords);

//---- 結果 ----
abc   def
["abc", "def"] (2)
