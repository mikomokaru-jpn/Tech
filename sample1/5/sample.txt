//--------------------------------------------------------------------------
// リテラルによる初期化
//--------------------------------------------------------------------------
NSNumber *num1 = [NSNumber numberWithInt:100]; //従来の方法
NSNumber *num2 = @200;                         //新しい方法

//--------------------------------------------------------------------------
// 変数による初期化
//--------------------------------------------------------------------------
int n = 300;
NSNumber *num3 = [NSNumber numberWithInt:n];   //従来の方法
NSNumber *num4 = @(n);                         //新しい方法

NSLog(@"%@ %@ %@ %@", num1, num2, num3, num4);

//整数に変換 
NSInteger integerValue = num1.integerValue;
