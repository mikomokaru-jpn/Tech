//------------------------------------------------------------------------------
// 初期化：クラスの混在可
//------------------------------------------------------------------------------
NSArray* array = @[@"安", @"伊", @"宇", @"江", @"於", @100, [NSDate date]];
NSArray* array = [NSArray arrayWithObjects:
                   @"安", @"伊", @"宇", @"江", @"於", @100, [NSDate date], nil];

for(id item in array){
    NSLog(@"%@", item);
} 

//------------------------------------------------------------------------------
// mutable array
//------------------------------------------------------------------------------
NSMutableArray *mutableArray = @[@1, @2, @3].mutableCopy;
mutableArray = [array mutableCopy];
