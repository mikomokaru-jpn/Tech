import Foundation
//----------------------------------------------------------------------
// Objective-c 整数 0〜9 の乱数を得る
//----------------------------------------------------------------------
int num = (int)arc4random_uniform(10);
NSLog(@"%d", num);    

//----------------------------------------------------------------------
// Swift 整数 0〜9 の乱数を得る
//----------------------------------------------------------------------
let num = Int(arc4random_uniform(UInt32(10)))
print(num)
