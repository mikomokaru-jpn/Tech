func clearLines(){    //self.lineAttrList = [LineAttribute]() //全部消去    //クリア対象の点を特定する    for i in 0 ..< lineAttrList.count{        for j in 0 ..< lineAttrList[i].pointList.count{            if let path = self.rectShape.path{                if path.boundingBox.isInside(lineAttrList[i].pointList[j]){                    //消去する点の座標を負値とする                    lineAttrList[i].pointList[j] = CGPoint.init(x: -1, y: -1)                }            }                        }    }    //線の集合の再作成    var tempLineAttrList = [LineAttribute]()    for line in lineAttrList{        var tempPointList = [CGPoint]()        for point in line.pointList{            if point.x < 0 {                //クリアされた点があれば、そこまでの点で一つの線とする                if tempPointList.count > 0 {                    var newLine = line                    newLine.pointList = tempPointList                    tempLineAttrList.append(newLine)                    tempPointList = [CGPoint]()                }            }else{                tempPointList.append(point)            }        }        if tempPointList.count > 0 {            var newLine = line            newLine.pointList = tempPointList            tempLineAttrList.append(newLine)            tempPointList = [CGPoint]()        }    }    lineAttrList = tempLineAttrList    self.needsDisplay = true}
