<?php/* 指定した日付の範囲の血圧レコードを読み込む *///DB接続require_once("init.php");require_once("MYDB.php");$pdo = db_connect();//JSONデータの読込み$json_post = file_get_contents('php://input');$json_string = mb_convert_encoding($json_post, 'UTF8', 'ASCII,JIS,UTF-8,EUC-JP,SJIS-WIN');$arr = json_decode($json_string, true);// SQLの実行 try{    $sql = "SELECT id, date, upper, lower, upper - lower AS pulse               FROM blood_pressure WHERE  ( id=:id ) AND ( date >=:fd AND date <=:td )  AND (confirm=1); ";    $stmh = $pdo->prepare($sql);    $stmh->bindValue(':id', $arr['id'], PDO::PARAM_INT);	    $stmh->bindValue(':fd', $arr['from_date'], PDO::PARAM_INT);    $stmh->bindValue(':td', $arr['to_date'], PDO::PARAM_INT);    $stmh->execute();}catch (PDOException $e) {    die('エラー:'. $e->getMessage());}//JSON形式で出力header('Content-type: application/json');echo json_encode($stmh->fetchAll(PDO::FETCH_ASSOC));?>
