//初期処理-(void)initialImage{    ....    //画像と同じ大きさのハイライト矩形を作成する。    //@property CALayer *highlightLayer;     _highlightLayer = [[CALayer alloc] init];      //オブジェクトの作成    _highlightLayer.frame = self.imageLayer.frame; //親画像のframeを参照    _highlightLayer.backgroundColor = [NSColor yellowColor].CGColor;     _highlightLayer.opacity = 0.4;                                   }//マウスダウン- (void)mouseDown:(NSEvent *)event{    ....    [self.imageLayer addSublayer:_highlightLayer];  //ハイライトする}//マウスアップ- (void)mouseUp:(NSEvent *)event{    ....    [_highlightLayer removeFromSuperlayer]; //ハイライトを消す}
