-(id)initWithFrame:(NSRect)frameRect{    if(![super initWithFrame:frameRect])        return nil;    self.wantsLayer = YES;    self.layer.backgroundColor = [NSColor blackColor].CGColor;    //ビューを作成しビュー配列に追加する    _viewList = [[NSMutableArray alloc] init];    NSSize vewSize = NSMakeSize(50.0, 50.0);    for(int i=0; i<10; i++){        NSString *num = [NSString stringWithFormat:@"%d", i];        UAView *aView = [[UAView alloc] initWithIndex:i number:num];        aView.frame = NSMakeRect(i*vewSize.height+2+(i*2), 2,                                  vewSize.width , vewSize.height);        aView.uaDelegate = self;        //デリゲート        [_viewList addObject:aView];    //ビューリストにビューを追加する        [self addSubview:aView];        //自ビューにビューを追加する    }    self.needsDisplay = YES;    return self;}
