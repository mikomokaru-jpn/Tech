-(id)initWithNumber:(NSInteger)num string:(NSString*)str{    if(![super init])        return nil;    //プロパティの設定    _number = num;    _string = str;    _font = [NSFont fontWithName:@"YuKyo-Bold" size:40];    //背景色    _highlightTo = NO;    _highlightFrom = NO;    self.wantsLayer = YES;    //ドラッグの受け入れオブジェクトの宣言    [self registerForDraggedTypes:@[@"public.text"]];    return self;}
