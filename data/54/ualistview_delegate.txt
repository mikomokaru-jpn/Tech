-(void)resetImageListTo:(NSInteger)toNumber from:(NSInteger)fromNumber{    NSString *temp = _viewList[fromNumber].number;    //添字の増分値    NSInteger incr;    if(toNumber == fromNumber){        return;    }else if (toNumber < fromNumber){        incr = -1;        for(NSInteger i=fromNumber; i > toNumber; i+=incr){            //文字列の移動            _viewList[i].number = _viewList[i+incr].number;            _viewList[i].needsDisplay = YES;        }    }else{        incr = 1;        for(NSInteger i=fromNumber; i < toNumber; i+=incr){            //文字列の移動            _viewList[i].number = _viewList[i+incr].number;            _viewList[i].needsDisplay = YES;        }    }    _viewList[toNumber].number = temp;    _viewList[toNumber].needsDisplay = YES;}
