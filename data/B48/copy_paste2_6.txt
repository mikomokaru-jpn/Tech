func menuWillOpen(_ menu: NSMenu){    menu.item(withTitle: "Paste")?.isEnabled = false    //貼り付けるイメージがペーストボードにあるか    if pasteboard.canReadItem(withDataConformingToTypes: [typeTIFF]){        menu.item(withTitle: "Paste")?.isEnabled = true        self.objectType = typeTIFF    }    //コピー/切り取りするイメージが表示中か    if let view = self.window.firstResponder as? UAView{        if view.image != nil{            menu.item(withTitle: "Cut")?.isEnabled = true            menu.item(withTitle: "Copy")?.isEnabled = true        }else{            menu.item(withTitle: "Cut")?.isEnabled = false            menu.item(withTitle: "Copy")?.isEnabled = false        }    }}
