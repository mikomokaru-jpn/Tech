//---- クラス宣言 ----#import "AClass.h"@interface BClass : NSObject<NSCopying>@property AClass *object;@property NSMutableArray<AClass*> *array;@end//---- 実装 ----@implementation BClass- (id)copyWithZone:(NSZone *)zone{    BClass* aCopy = [[[self class] alloc] init];    if( aCopy ){          //カスタムクラス型のコピー        aCopy.object = [_object copy];         //カスタムクラス型の配列のコピー        aCopy.array = [[[_array class] alloc]                         initWithArray:_array copyItems:YES];     }    return aCopy;}@end
