//------------------------------------------------------------------------------
// プロトコル宣言
//------------------------------------------------------------------------------
@protocol CPViewDelegate
-(void)setMenuAvailable;
@end
#import <Cocoa/Cocoa.h>
//------------------------------------------------------------------------------
// インタフェース宣言
//------------------------------------------------------------------------------
@interface CPView : NSView
@property (weak, nonatomic) id<CPViewDelegate> cpDelegate;  //デリゲート変数
@end
//------------------------------------------------------------------------------
// クラス実装
//------------------------------------------------------------------------------
#import "CPView.h"

@implementation CPView

//マウスの右クリック：コンテキストメニューを表示
- (void)rightMouseDown:(NSEvent *)theEvent{
    [_cpDelegate setMenuAvailable];     //メニューの編集（有効化）
    [super rightMouseDown:theEvent];
}

@end
