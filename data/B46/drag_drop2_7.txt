override func performDragOperation(_ info: NSDraggingInfo) -> Bool{    //オブジェクトの存在チェック    if !info.draggingPasteboard().canReadItem(withDataConformingToTypes:                                               [typeTIFF]){        print("image not found")        return false    }    //イメージを読み込む    if let data = info.draggingPasteboard().data(forType: .tiff){        self.image = NSImage.init(data: data)        //ドロップ成功        self.needsDisplay = true        return true    }            return false //ドロップ不成立}
