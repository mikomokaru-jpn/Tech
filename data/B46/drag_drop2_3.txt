override func mouseDown(with event: NSEvent) {    guard let image = self.image else {        return //イメージが表示されていないので何もしない    }    //ドラッグするオブジェクトを配列に格納する    var itemList = [NSDraggingItem]()    itemList.append(NSDraggingItem.init(pasteboardWriter: image))    //ドラッグ中にスクリーンに表示するイメージの作成    itemList[0].setDraggingFrame(self.bounds, contents: image)    //ドラッグの開始の宣言    self.beginDraggingSession(with: itemList, event: event, source: self)    }
