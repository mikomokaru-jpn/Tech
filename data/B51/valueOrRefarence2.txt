//関数定義　構造体、配列の参照を引数とする（inoutパラメータ）func updateReference( record: inout Record, array: inout [String]){    record.array[0] = 999    array[1] = "い"}//変数の作成var rec = Record.init(array: [1,2,3],                      text: "Hello",                      value: 1.34)var array = ["A", "B", "C"]//関数を呼び出す　引数には変数のポインタ(&)を渡すupdateReference(record: &rec, array: &array)print(rec.array) -> [999, 2, 3]print(array)     -> ["A", "い", "C"]/* 値渡しの引数そのものは更新はできないと、コンパイルエラーではじかれる。func updateValue(record: Record, array: [String]){    record.array[0] = 999    array[1] = "い"}*/
