//----------------------------------------------------------------------// 構造体・値のコピー//----------------------------------------------------------------------struct Record{    var array: [Int]    var text: String    var value: Float}//変数の作成var rec1 = Record.init(array: [1,2,3],                      text: "Hello",                      value: 1.34)var rec2 = rec1         //*** 変数のコピー rec1.text = "こんにちは"  //*** コピー元の変更 print(rec1.text) -> "こんにちは"print(rec2.text) -> "Hello"//----------------------------------------------------------------------// 配列・値のコピー//----------------------------------------------------------------------var array1 = ["A", "B", "C"]var array2 = array1   //*** オブジェクトのコピーarray1[0] = "あ"      //*** コピー元の変更 print(array1) -> ["あ", "B", "C"]print(array2) -> ["A", "B", "C"]//----------------------------------------------------------------------// クラスオブジェクト・参照のコピー//----------------------------------------------------------------------class AClass{    var array: [Int]    var text: String    var value: Float    //イニシャライザ    init(_ array:[Int], _ text:String, _ value:Float) {        self.array = array        self.text = text        self.value = value    }}//オブジェクトの作成var aObj1 = AClass.init([1,2,3], "Hello", 1.34)var aObj2 = aObj1         //*** オブジェクトのコピーaObj1.text = "こんにちは"   //*** コピー元の変更print(aObj1.text) -> "こんにちは"print(aObj2.text) -> "こんにちは"
