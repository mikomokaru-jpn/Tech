//出力テキストNSString *text =  @"つれづれなるまゝに、日くらし硯に向かひて、"   "心にうつりゆくよしなしごとをそこはかとなく書き付くれば、"   "あやしうこそ物狂ほしけれ。";//ファイルパスNSString *filePath = [NSString stringWithFormat:@"%@%@",                      NSHomeDirectory(),                      @"/Documents/output.txt"];//NSStringオブジェクトをファイルに書き出すNSError *error;if(![text writeToFile:filePath           atomically:YES             encoding:NSUTF8StringEncoding                error:&error]){    NSLog(@"!!! %ld !!! %@",error.code, error);}
