//出力テキストNSString *text =  @"つれづれなるまゝに、日くらし硯に向かひて、"   "心にうつりゆくよしなしごとをそこはかとなく書き付くれば、"   "あやしうこそ物狂ほしけれ。";//NSStringオブジェクトをNSDataオブジェクトに変換するNSData *data = [text dataUsingEncoding:NSUTF8StringEncoding];//ファイルパスNSString *filePath = [NSString stringWithFormat:@"%@%@",                      NSHomeDirectory(),                      @"/Documents/output.txt"];NSURL *fileURL = [NSURL fileURLWithPath:filePath];//NSDataオブジェクトをファイルに書き出すNSError *error;if (![data writeToURL:fileURL atomically:YES]){    NSLog(@"!!! %ld !!! %@",error.code, error);}
