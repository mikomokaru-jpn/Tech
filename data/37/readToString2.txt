//ファイルパスNSString *filePath = [NSString stringWithFormat:@"%@%@",                      NSHomeDirectory(),                      @"/Documents/input.txt"];NSURL *fileURL = [NSURL fileURLWithPath:filePath];//ファイルを読むNSString *contents;NSFileManager *fm = [NSFileManager defaultManager];if ([fm fileExistsAtPath:filePath]){    NSError *error;    contents = [NSString stringWithContentsOfURL:fileURL                           encoding:NSUTF8StringEncoding                                            error:&error];    if (error != nil)        NSLog(@"!!! %ld !!! %@",error.code, error);}else{    NSLog(@"File not found");}//ファイルの中身を表示するNSLog(@"%@",contents);
