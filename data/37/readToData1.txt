//ファイルパスNSString *filePath = [NSString stringWithFormat:@"%@%@",                      NSHomeDirectory(),                      @"/Documents/input.txt"];//ファイルを読むNSData *data;NSFileManager *fm = [NSFileManager defaultManager];if ([fm fileExistsAtPath:filePath]){    data = [NSData dataWithContentsOfFile:filePath];}else{    NSLog(@"File not created");} //NSSreingオブジェクトに変換するNSString *contents = [[NSString alloc] initWithData:data                       encoding:NSUTF8StringEncoding];//ファイルの中身を表示するNSLog(@"%@",contents);
