//ファイルパスNSString *filePath = [NSString stringWithFormat:@"%@%@",                      NSHomeDirectory(),                      @"/Documents/input.txt"];NSURL *fileURL = [NSURL fileURLWithPath:filePath];//ファイルを読むNSData *data;NSFileManager *fm = [NSFileManager defaultManager];if ([fm fileExistsAtPath:filePath]){    //テキストファイルを読み込む    data = [NSData dataWithContentsOfURL:fileURL];}else{    NSLog(@"File not created");}NSString *contents = [[NSString alloc] initWithData:data                           encoding:NSUTF8StringEncoding];//ファイルの中身を表示するNSLog(@"%@",contents);
